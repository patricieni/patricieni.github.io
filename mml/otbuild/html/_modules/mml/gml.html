[docs]def algorithm1_similar(M, w, G_all, T_all, k=3):
    """Algorithm 1 from Cuturi GML for the similar neighbours
    
    Args:
        M (matrix): Ground metric
        w (matrix): Similarity Matrix
        G_all (matrix): All Wasserstein distances
        T_all (tensor): All coupling matrices
        k (int, optional): Defaults to 3. Neighbours to consider
    
    Returns:
        z_plus, gamma_plus: The weighted sum for similar components for the minimization criteria (z) and the gradients (gamma)
    """
    
    # Select the k smallest similar/dissimilar neighbours for each histogram i and the elements that come with it, i.e. w, G and X*, i.e. Wasserstein distance
    S_similar = similar_neighbours(k,w,G_all)
    grad_similar = gradient_neighbours(w,T_all, S_similar)
    
    #print("All done with selecting neighbours and gradients") 
    
    # Compute sum for each individual histogram and return a list of sums 
    similar_sum = sum_neighbours(S_similar)
    grad_similar_sum = sum_gradients(grad=grad_similar)
    
    #print("Sum all histogram information")
    # Rename them as in GML for clarity and sum the objective and gradients across all histograms for similar/dissimilar pairs (sum the lists)
    z_plus = np.sum(similar_sum)
    gamma_plus = np.sum(grad_similar_sum,axis=0)

    # Return a (d 2) vector 
    gamma_plus = helper.triu_vector(gamma_plus) + helper.tril_vector(gamma_plus)
    
    # Return a semi-metric matrix with zero on the diagonal
    #gamma_plus = helper.triu(gamma_plus) + helper.tril(gamma_plus)

    
    return z_plus, gamma_plus

    """Algorithm 1 from Cuturi GML for dissimilar neighbours
    
    Args:
        M (matrix): Ground metric
        w (matrix): Similarity Matrix
        G_all (matrix): All Wasserstein distances
        T_all (tensor): All coupling matrices
        k (int, optional): Defaults to 3. Neighbours to consider
    
    Returns:
        z_plus, gamma_plus: The weighted sum for dissimilar components for the minimization criteria (z) and the gradients (gamma)
    """
    # Select the k smallest similar/dissimilar neighbours for each histogram i and the elements that come with it, i.e. w, G and X*, i.e. Wasserstein distance
    S_dissimilar = dissimilar_neighbours(k,w,G_all)
    grad_dissimilar = gradient_neighbours(w,T_all, S_dissimilar)
    
    #print("All done with selecting neighbours and gradients")
    
    # Compute sum for each individual histogram and return a list of sums 
    dissimilar_sum = sum_neighbours(S_dissimilar)
    grad_dissimilar_sum = sum_gradients(grad=grad_dissimilar)
    
    #print("Sum all histogram information")
    # Rename them as in GML for clarity and sum the objective and gradients across all histograms for similar/dissimilar pairs (sum the lists)
    z_minus = np.sum(dissimilar_sum)
    gamma_minus =  np.sum(grad_dissimilar_sum, axis=0)

    # Return a (d 2) vector 
    gamma_minus =  helper.triu_vector(gamma_minus) + helper.tril_vector(gamma_minus)

    # Return a semi-metric matrix with zero on the diagonal
    #gamma_minus = helper.triu(gamma_minus)+ helper.tril(gamma_minus)
    
    return z_minus, gamma_minus

[docs]def similar_neighbours(neigh, w, G_all):
    """ Find similar neighbours based on Wasserstein distances and a similarity matrix
    
    Args:
        neigh ([type]): Number of similar neighbours to look at 
        w ([type]): Similarity matrix
        G_all ([type]): Wasserstein distances between each pair of histograms 
    
    Returns:
       list(dict(i,w,g)) A list of all histograms with their associated information on k similar neighbours (dictionary of (w,G) pair)
    """

    # Use w and G to find the right similar(plus) neighbours
    # Sort and return index, this way you can access the right stuff 
    n  = w.shape[0]
    S_plus = []
    # Loop over all histograms - one row has all neighbours of that row. 
    for i in range(n): 
        # Sort W distances for ith row (or histogram) in ascending order and retrieve indices
        indices = np.argsort(G_all[i,:],kind='mergesort')
        n_plus = {}
        j = 1 # Always start at 1 since first position will be j==i. Distance to itself doesn't count
        while (len(n_plus) < neigh) and (j!=n):
            index = indices[j]
            if w[i,index] > 0: 
                g = G_all[i,index] # G_ij(M)
                sim = w[i,index] # w_ij > 0
                n_plus.update({index:np.array([g,sim])})
                j+=1 
            else:
                j+=1 
        S_plus.append(n_plus)
    return S_plus

[docs]def sum_neighbours(S):
    """Sums up similar neighbours of all elements in the dataset. 
    This amounts to sum (S^+_{ik}(M)) from Cuturi GML
    
    Args:
        S (list): List of similar/dissimilar neighbours with pairs (w,G)
    
    Returns:
        (list): The weighted sum of distances to similar and disimilar neighbours
    """
    # S is a  list of n dictionaries containing pairs of positive neighbours for that histogram. 
    # To compute the sum: 
    S_sum = []
    for el in S: 
        el_sum = 0
        for key in el: 
            el_sum += el[key][0] * el[key][1]
        S_sum.append(el_sum)
    
    return S_sum

[docs]def gradient_neighbours(w,T_all,S):
    """ Finds all gradients for the subset of neighbours as a list of OT matrices 

    Args:
       w (matrix): Similarity matrix
       T_all (tensor): Tensor of al optimal matrices given by all pairs of histograms
       S (list): List containing all information similar OR dissimilar neighbours
   
    Returns:
       list: A list of all OT matrices (~gradients of G) for histograms 
       containing k similar/dissimilar neighbours (dictionary of (w,X) pair)
    """
    # Use w and G to find the right plus/minus neighbours
    # Sort and return index, this way you can access the right stuff 
    n = len(S)
    i = 0
    grad_neigh = []
    for el in S: 
        neighbours = {}
        for key in el: 
            w = el[key][1] # Take w from the dictionary of the distances
            T = T_all[i][key] # Take T from the computation you have
            neighbours.update({key:np.array([T,w])})
        grad_neigh.append(neighbours)    
        i+=1
    return grad_neigh



[docs]def sum_gradients(grad):
    """Sums up neighbours of all gradients in the dataset. 
    This amounts to \grad(S^+_{ik}(M)) from Cuturi GML section 4.1
    
    Args:
        grad ([type]): List of neighbour gradients, similar/dissimilar
    
    Returns:
        float: A list of the weighted sum of wasserstein distance gradients(equivalent to OT matrix) to similar neighbours
    for all histograms
    """

    # S_plus is a  list of n dictionaries containing pairs of positive neighbours for that histogram. 
    # To compute the sum: 
    grad_sum = []
    for el in grad: 
        el_sum = 0
        for key in el: 
            el_sum += el[key][0] * el[key][1]
        grad_sum.append(el_sum)
    return grad_sum

[docs]def similarity_matrix(X, Y, k):
    """Builds a similarity matrix using X and Y labels as well as k neighbours. Similarity is given by same labels and the similarity value is proportional to size of data n and neighbours k.
    The opposite is true for dissimilarity, same value with a minus sign.
        
    Args:
        X (array(float)): Dataset
        Y (array(float)): Labels
        k (int): Number of neighbours used in GML

    Returns:
        array[float,float]: A matrix w of shape X.shape[0],X.shape[0]
    """

    n = X.shape[0]
    w = np.zeros(shape=(n, n))
    for i in range(n):
        for j in range(n):
            if (i != j) & (Y[i] == Y[j]):
                w[i][j] = 1/(n*k)
            elif (i != j) & (Y[i] != Y[j]):
                w[i][j] = -1/(n*k)
            else:
                w[i][j] = 0

    return w

    """Projects matrices that are not semi-metric to the closest true semi-metric using the metricNearnes software
    TODO: Add in the C++ code the possibility to pass some sort of byte array to the metric code, currently reading and 
    writing to a txt file "metricNearness_process/matrix"
    
    Args:
        m_initial (array): Approximate metric to project
        filepath (string): Filename to read and write the matrix 
    
    Returns:
        array: Projection of m_initial using the metricNearness software
    """

    path = os.path.join(os.getcwd(),"metricNearness_process")
    os.chdir(path)
    
    first_line = str(m_initial.shape[0]) +" "+str(m_initial.shape[1])
    
    with open(f, 'w') as outfile:
        outfile.write(first_line+"\n")
        np.savetxt(f, np.matrix(m_initial))
    
    output = subprocess.check_output('./metric -a 0 -f '+f+ ' -m 10', shell = True )
    lines = str(output).split('\\n')
    shape = lines[5].split(" ")
    rows = int(shape[0])
    cols = int(shape[1])
    # Although its square matrix assume it's not
    matrix = lines[6:6+rows]
    m  = np.zeros((rows,cols))

    for i,row in enumerate(matrix): 
        vals = row.strip().split(" ")
        m[i] = np.array(vals,dtype=float)
     
    with open(f, 'w') as outfile:
        outfile.write(first_line+"\n")
        np.savetxt(f, np.matrix(m))
        #fmt="%.3f"
    path = Path(os.getcwd()).parent
    os.chdir(path)   
    return m
