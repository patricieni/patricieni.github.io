[docs]def positive_and_normalize(x_train,x_test,norm):
    """Normalize the data and if negative, make it positive before normalizing by adding the min value needed
    
    Args:
        x_train (array): Training data
        x_test (array): Test data
        norm (str): Either L1 or L2, or something else perhaps
    
    Returns:
        list(array, array): Positive and normalized train and test data
    """
    x_train_positive = x_train
    x_test_positive = x_test

    if (x_train < 0).sum() !=0: 
        for i in range(x_train.shape[0]):
            x_train_positive[i] += np.abs(x_train[i].min())

    if (x_test < 0).sum() !=0: 
        for i in range(x_test.shape[0]):
            x_test_positive[i] += np.abs(x_test[i].min())

    if (norm.lower() != 'l1' or norm.lower() != 'l2'):
        norm = 'l1'
    # Normalize but add norm too in the future
    row_sum = x_train_positive.astype(np.float64).sum(axis=1)
    row_sum_test = x_test_positive.astype(np.float64).sum(axis=1)

    X_train_normalized = np.true_divide(x_train_positive,row_sum[:,np.newaxis])
    X_test_normalized = np.true_divide(x_test_positive,row_sum_test[:,np.newaxis])

    return X_train_normalized,X_test_normalized

        x_train (array): Training data
        x_test (array): Test data
        norm (str): Either L1 or L2, or something else perhaps
    
    Returns:
        list(array, array): Normalized train and test data
    """

    if (norm.lower() != 'l1' or norm.lower() != 'l2'):
        norm = 'l1'
        
    sc = Normalizer(norm).fit(x_train)
    X_train_normalized = sc.transform(x_train)
    X_test_normalized = sc.transform(x_test)
    return [X_train_normalized, X_test_normalized]
